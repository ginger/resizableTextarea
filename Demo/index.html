resizableTextarea v.1.1 for Mootools 1.2

Vertical and horizontal

var textarea = new resizableTextarea($$("div.textarea"), {
	handler: ".handler",
	modifiers: {x: true, y: true},
	size: {x:[100, 600], y:[50, 300]},
	onResize: function(current) {
		$("size").set("html", current.getWidth() + " " + current.getHeight());
	}
});
 

Vertical only

var textarea2 = new resizableTextarea($$("div.textarea2"), {
	handler: ".handler",
	modifiers: {x: false, y: true},
	size: {y:[50, 300]}
});

Horizontal only

var textarea3 = new resizableTextarea($$("div.textarea3"), {
	handler: ".handler",
	modifiers: {x: true, y: false},
	size: {x:[100, 600]}
});

Vertical only + flexible
